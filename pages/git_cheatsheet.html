Git Command Cheat Sheet

# Common Commands


# Initialize a repo
git init


# Clone a repo
git clone URL


# Stage and commit
git add .
git commit -m "message"


# Push and pull
git push


# Branching
git branch feature/x
git checkout feature/x
git checkout main
git merge feature/x


# History & Fixes
git log --oneline
git revert 
git reset --hard
